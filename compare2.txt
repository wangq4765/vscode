apiVersion: v1
kind: Pod
metadata:
  name: nginx-demo
  labels:
    type: app
    test: 1.0.0
  namespace: 'default'
spec:
  containers:
  - name: nginx
    image: nginx:1.7.9
    imagePullPolicy: IfNotPresent
    startupProbe: 
      httpGet: 
        path: /api/path
        port: 80
      failureThreshold: 5
      periodSeconds: 10
      successThreshold: 1
      timeoutSeconds: 1
    command:
    - nginx
    - -g
    - 'daemon off;'
    workingDir: /usr/share/nginx/html
    ports:
    - name: http
      containerPort: 80
      protocol: TCP
    env:
    - name: JVM_OPTS
      value: '-Xms128m -Xmx128m'
    resources:
      requests:
        cpu: 100m
        memory: 128Mi
      limits:
        cpu: 200m
        memory: 256Mi
  restartPolicy: OnFailure
